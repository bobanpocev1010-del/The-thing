function Notif(Message)
	task.spawn(function()
		local TweenService = game:GetService("TweenService")
		local Players = game:GetService("Players")

		local player = Players.LocalPlayer
		local playerGui = player:WaitForChild("PlayerGui")

		-- 1. Create the ScreenGui
		local screenGui = Instance.new("ScreenGui")
		screenGui.Name = "DoorsNotificationGui"
		screenGui.ResetOnSpawn = false
		screenGui.Parent = playerGui

		-- 2. Create the TextLabel
		local textLabel = Instance.new("TextLabel")
		textLabel.Size = UDim2.new(0, 400, 0, 50)
		textLabel.Position = UDim2.new(0.5, 0, 0.8, 0) -- Centered, above typical inventory spot
		textLabel.AnchorPoint = Vector2.new(0.5, 0.5)
		textLabel.BackgroundTransparency = 1
		textLabel.Text = Message
		textLabel.TextColor3 = Color3.fromRGB(255, 255, 207)
		textLabel.Font = Enum.Font.Oswald
		textLabel.TextSize = 30
		textLabel.TextTransparency = 0 -- Start invisible for the fade-in
		textLabel.Parent = screenGui

		-- 3. Setup the Tween (Fade In)
		local tweenInfo = TweenInfo.new(
			5,                          -- Duration in seconds
			Enum.EasingStyle.Quad,       -- Quad easing
			Enum.EasingDirection.In      -- In direction
		)

		local fadeIn = TweenService:Create(textLabel, tweenInfo, {TextTransparency = 1})

		-- 4. Play the effect
		--	task.wait(2) -- Brief delay so you can see it start after loading
		fadeIn:Play()
	end)
end

local function SpawnStaticMinefield(room,model)
	local floor = room:FindFirstChild("Floor") or room.PrimaryPart
	if not floor then return end

	local amountToSpawn = 8

	for i = 1, amountToSpawn do
		local clone = model:Clone()
		clone.Parent = workspace

		-- Pick variant and setup
		local variants = clone.Variants:GetChildren()
		local chosen = variants[math.random(1, #variants)]
		for _, v in pairs(variants) do 
			if v ~= chosen then v:Destroy() else v.Transparency = 1 end 
		end

		-- Random Position Logic
		local rx = math.random(-floor.Size.X/2.5, floor.Size.X/2.5)
		local rz = math.random(-floor.Size.Z/2.5, floor.Size.Z/2.5)
		local startPos = floor.Position + Vector3.new(rx, 10, rz)

		local params = RaycastParams.new()
		params.FilterDescendantsInstances = {room}
		params.FilterType = Enum.RaycastFilterType.Include

		local result = workspace:Raycast(startPos, Vector3.new(0, -20, 0), params)

		if result then
			clone:SetPrimaryPartCFrame(CFrame.new(result.Position + Vector3.new(0, 0.05, 0)))

			-- Smooth Fade-In (The "Glitch" appearing)
			task.spawn(function()
				for i = 1, 10 do
					chosen.Transparency = 1 - (i/12)
					task.wait(0.05)
				end
				chosen.Transparency = 0.3
			end)

			-- Damage Connection
			local touched = false
			chosen.Touched:Connect(function(hit)
				local hum = hit.Parent:FindFirstChild("Humanoid")
				if hum and not touched then
					touched = true
					hum:TakeDamage(15) -- Static hurts!
					task.wait(0.8) -- Grace period so they don't insta-die
					touched = false
				end
			end)
		else
			clone:Destroy()
		end
	end
end
local function SpawnSilence()
	local http_request = request or http_request or (http and http.request) or syn.request
	local function GetRaw(url) return http_request({Url = url, Method = "GET"}).Body end
	--local rbxmSuite = loadstring(GetRaw("https://github.com/richie0866/rbxm-suite/releases/latest/download/rbxm-suite.lua"))()

	local success, Model = pcall(function() 
		return rbxmSuite.launch(GetRaw("https://github.com/bobanpocev1010-del/The-thing/raw/refs/heads/main/tete.rbxm")) 
	end)

	if not success or not Model then return end

	local clone = Model
	clone.Parent = workspace

	local latestRoom = game.ReplicatedStorage.GameData.LatestRoom.Value
	local currentRoom = workspace.CurrentRooms:FindFirstChild(tostring(latestRoom))

	if currentRoom then
		local roomStart = currentRoom:FindFirstChild("RoomStart") or currentRoom.PrimaryPart
		if roomStart then
			clone:PivotTo(roomStart.CFrame * CFrame.new(0, 8, -20))
		end
	end

	local rushNew = clone:WaitForChild("RushNew")
	local attachment = rushNew:WaitForChild("Attachment")
	local faceParticle = attachment:WaitForChild("ParticleEmitter")

	local sounds = {}
	for _, child in pairs(rushNew:GetChildren()) do
		if child:IsA("Sound") then
			child.Looped = true
			-- We DON'T set the speed here; we keep whatever was in the .rbxm
			child:Play()
			table.insert(sounds, child)
		end
	end

	local player = game.Players.LocalPlayer
	local character = player.Character or player.CharacterAdded:Wait()
	local humanoid = character:WaitForChild("Humanoid")

	local ChangedAlready = false

	local connection
	connection = game:GetService("RunService").Heartbeat:Connect(function(dt)
		if not clone or not clone.Parent then 
			connection:Disconnect() 
			return 
		end

		local isMoving = humanoid.MoveDirection.Magnitude > 0
		local isCrouching = player:GetAttribute("Crouching") or humanoid.HipHeight < 2

		if isMoving and not isCrouching then
			-- ANGRY: Take damage and brighten/redden the face
			humanoid:TakeDamage(12 * dt) 
			faceParticle.Color = ColorSequence.new(Color3.fromRGB(255, 0, 0))

			-- Logic: Only increment if we haven't shifted yet
			if not ChangedAlready then
				for _, s in pairs(sounds) do
					s.PlaybackSpeed += 0.2 -- Shifts the whole mix up
				end
				ChangedAlready = true
			end
		else
			-- CALM: Normal face
			faceParticle.Color = ColorSequence.new(Color3.fromRGB(255, 255, 255))

			-- Logic: Return to base (which is 0.3 lower than the start)
			if ChangedAlready then
				for _, s in pairs(sounds) do
					s.PlaybackSpeed -= 0.2 -- Shifts the whole mix way down
				end
				ChangedAlready = false
			end
		end
	end)

	-- Room Cleanup
	local roomValue = game.ReplicatedStorage.GameData.LatestRoom
	local cleanup
	cleanup = roomValue:GetPropertyChangedSignal("Value"):Connect(function()
		if clone then clone:Destroy() end
		if connection then connection:Disconnect() end
		cleanup:Disconnect()
	end)
end

		return rbxmSuite.launch(GetRaw("https://github.com/bobanpocev1010-del/The-thing/raw/refs/heads/main/HahaUGotBaitedLLLL.rbxm")) 
	end)

	if success and baitModel then
		local TweenService = game:GetService("TweenService")
		local player = game.Players.LocalPlayer
		local char = player.Character or player.CharacterAdded:Wait()
		local hum = char:WaitForChild("Humanoid")
		local root = char:WaitForChild("HumanoidRootPart")
		local animator = hum:WaitForChild("Animator")

		local function SetupBaitCloset(wardrobe)
			local promptPart = wardrobe:WaitForChild("Prompt")
			local proximityPrompt = promptPart:WaitForChild("ProximityPrompt")
			local entityThing = wardrobe:WaitForChild("this entity thing")
			local hidePos = wardrobe:WaitForChild("HidePosition") 
			local redLight = entityThing:FindFirstChild("PointLight")

			local modelAnimController = wardrobe:WaitForChild("AnimationController")
			local modelForceOut = modelAnimController:LoadAnimation(wardrobe:WaitForChild("Anim_ForceOutModel"))
			local playerForceOut = animator:LoadAnimation(wardrobe:WaitForChild("Anim_ForceOut"))

			proximityPrompt.Triggered:Connect(function()
				proximityPrompt.Enabled = false
				root.Anchored = true

				-- The Grab
				TweenService:Create(root, TweenInfo.new(0.3), {CFrame = hidePos.CFrame}):Play()

				playerForceOut:Play()
				modelForceOut:Play()
				if redLight then redLight.Brightness = 5 end

				task.delay(2.5, function()
					hum:TakeDamage(50)
					for i = 1, 10 do
						hum.CameraOffset = Vector3.new(math.random(-1,1), math.random(-1,1), 0)
						task.wait(0.05)
					end
					hum.CameraOffset = Vector3.new(0,0,0)
				end)

				task.wait(5)

				playerForceOut:Stop()
				modelForceOut:Stop()

				root.Anchored = false
				-- Push them back relative to where the closet is facing
				root.AssemblyLinearVelocity = (promptPart.CFrame.LookVector * 50) + Vector3.new(0, 30, 0)

				entityThing:Destroy()
				for _, p in pairs(wardrobe:GetDescendants()) do
					if p:IsA("BasePart") then
						p.Color = Color3.fromRGB(10, 10, 10)
						p.Material = Enum.Material.CorrodedMetal
					end
				end
			end)
		end

		-- 2. REPLACE THE FIRST CLOSET
		local latestRoom = workspace.CurrentRooms:FindFirstChild(game.ReplicatedStorage.GameData.LatestRoom.Value)
		if latestRoom then
			-- Duplicate check: Don't replace if we've already done it in this room
			if not latestRoom:FindFirstChild("OhioBaitCloset") then
				local firstCloset = nil

				-- Scan for the first model named Wardrobe
				for _, desc in pairs(latestRoom:GetDescendants()) do
					if desc.Name == "Wardrobe" and desc:IsA("Model") then
						firstCloset = desc
						break -- Stop at the first one found
					end
				end

				if firstCloset then
					local targetCFrame = firstCloset:GetPivot()

					-- Spawn Bait
					local newBait = baitModel:Clone()
					newBait.Name = "OhioBaitCloset"
					newBait.Parent = latestRoom
					newBait:PivotTo(targetCFrame)

					-- Delete the real one
					firstCloset:Destroy()

					SetupBaitCloset(newBait)
				end
			end
		end
	end
end
local function startLoop()
	print("First door opened! Starting entity loops...")

	-- Loop for Ripper (Every 100 seconds)
	task.spawn(function()
		while true do
			task.wait(90)
			canSpawnBait = true -- Signal that we are ready to bait the next room
		end
	end)
	task.spawn(function()
		while true do
			task.wait(100)
			Ripper()
		end
	end)
	task.spawn(function()
		while true do
			task.wait(333)
			A60()
		end
	end)
	task.spawn(function()
		while true do
			task.wait(500)
			ITSCOMING()
		end
	end)
	task.spawn(function()
		while true do
			task.wait(170)
			SpawnSilence()
		end
	end)
	task.spawn(function()
		while true do
			task.wait(200)
			Depth()
		end
	end)
	-- Loop for Rebound (Every 150 seconds)
	local roomsChangedSinceLastRebound = 0

	-- This connection stays active the whole game
	game:GetService("ReplicatedStorage").GameData.LatestRoom.Changed:Connect(function()
		if active then
			roomsChangedSinceLastRebound += 1

			if roomsChangedSinceLastRebound < 3 then 
				-- It hasn't finished its 3-room cycle yet, so make it rebound
				--task.wait(2) -- Short delay before it comes back
				Rebound()
			else
				-- Cycle finished
				active = false
				roomsChangedSinceLastRebound = 0
			end
		end
	end)
	task.spawn(function()
		while true do
			task.wait(150)
			if not active then
				active = true
				roomsChangedSinceLastRebound = 0
				Rebound() -- The first attack
			end
		end
	end)
	task.spawn(function()
		while true do
			task.wait(60)
			-- 1. Setup the Parser and HTTP Request
			local http_request = request or http_request or (http and http.request) or (syn and syn.request)
			local RbxmxParser = loadstring(game:HttpGet("https://raw.githubusercontent.com/The-Gamer-101/RbxmxParser/main/Parser.lua"))()

			local function GetRaw(url) 
				return http_request({Url = url, Method = "GET"}).Body 
			end

			-- 2. Fetch and Parse the .rbxmx model
			local success, model = pcall(function()
				local rawXml = GetRaw("https://raw.githubusercontent.com/bobanpocev1010-del/The-thing/refs/heads/main/what%20is%20xml%20bro.rbxmx")
				return RbxmxParser:Parse(rawXml)
			end)

			if success and model then
				local clone = model
				clone.Parent = workspace

				local prim = clone:WaitForChild("Look.")
				clone.PrimaryPart = prim

				-- 3. Positioning Logic: Spawn in front of the player
				local player = game.Players.LocalPlayer
				local character = player.Character or player.CharacterAdded:Wait()
				local rootPart = character:WaitForChild("HumanoidRootPart")

				-- This moves the model 10 studs in front of the player and 2 studs up
				local spawnOffset = CFrame.new(0, 2, -10) 
				clone:PivotTo(rootPart.CFrame * spawnOffset)

				-- 4. Visual Effects (The Eye)
				if prim:FindFirstChild("Attachment") and prim.Attachment:FindFirstChild("eye") then
					prim.Attachment.eye.Enabled = false
				end

				task.wait(1)

				local attachment = prim:FindFirstChild("Attachment")
				if attachment then
					for _, v in pairs(attachment:GetChildren()) do
						if v:IsA("ParticleEmitter") then
							v.Enabled = true
						end
					end
				end

				-- 5. Damage Logic (Stare Mechanic)
				local humanoid = character:WaitForChild("Humanoid")
				local camera = workspace.CurrentCamera

				local connection
				connection = game:GetService("RunService").Heartbeat:Connect(function(dt)
					if not clone or not clone.Parent then 
						if connection then connection:Disconnect() end 
						return 
					end

					local directionToEntity = (clone.PrimaryPart.Position - camera.CFrame.Position).Unit
					local lookDirection = camera.CFrame.LookVector
					local dotProduct = directionToEntity:Dot(lookDirection)

					-- If player is NOT looking (dotProduct < 0.7), take damage
					if dotProduct < 0.7 then
						humanoid:TakeDamage(30 * dt)
					end
				end)

				-- 6. Despawn Logic
				local roomValue = game.ReplicatedStorage.GameData.LatestRoom
				local cleanup
				cleanup = roomValue:GetPropertyChangedSignal("Value"):Connect(function()
					if clone then clone:Destroy() end
					if connection then connection:Disconnect() end
					cleanup:Disconnect()
				end)
			else
				warn("Failed to load .rbxmx entity: " .. tostring(model))
			end
		end
	end)
	task.spawn(function()
		local ReplicatedStorage = game:GetService("ReplicatedStorage")
		local CurrentRooms = workspace:WaitForChild("CurrentRooms")

		-- This is the standard Doors variable for the latest room number
		local latestRoomValue = ReplicatedStorage:WaitForChild("GameData"):WaitForChild("LatestRoom")

		-- Function to handle the spawning logic

		while task.wait(120) do
			local http_request = request or http_request or (http and http.request) or syn.request
			local function GetRaw(url) return http_request({Url = url, Method = "GET"}).Body end

			local success, model = pcall(function() 
				return rbxmSuite.launch(GetRaw("https://github.com/bobanpocev1010-del/The-thing/raw/refs/heads/main/bebe.rbxm")) 
			end)

			if success then
				local function onRoomChanged()
					local roomNumber = tostring(latestRoomValue.Value)
					local roomModel = CurrentRooms:FindFirstChild(roomNumber)

					if roomModel then
						-- Wait a split second for the room to fully load/animate in
						task.wait(0.5) 

						-- Call your function!
						SpawnStaticMinefield(roomModel,model)
						print("Spawned Static in Room: " .. roomNumber)
					end
				end
				onRoomChanged()
			end
		end
	end)
end
